A minimum problem

Thursday, January 16, 2020, 8:00:00 am programming 　 math

A set SSS consists of 7 distinct integers x0<x1<⋯<x6x_0 < x_1 < \cdots < x_6x0​<x1​<⋯<x6​. If the set { x−y: x,y∈S,  x≠yx-y: \ x,y\in S, \ \ x \neq yx−y: x,y∈S,  x​=y } has 42 elements, find the minimum of x6−x0x_6-x_0x6​−x0​.

Generalize the problem. (***This is too hard for me...***)

Answer. 25.

Solution. Brute force is a good method. Create a Java package named minimum_problem, and then create two Java files.

File: Main.java

package minimum_problem;

/**
 * Main
 */
public class Main {

    public static void main(final String... args) {
        long time = System.currentTimeMillis();
        solver(100);
        System.out.println("\nTime elapsed: " + (System.currentTimeMillis() - time) + " milliseconds.\n");
    }

    /**
     * Gives the set D = {y - x : x, y are in {@code py_list} and x < y}.
     * 
     * @param py_list the list.
     * @return the set D.
     */
    public static PyList difference(final PyList py_list) {
        final PyList result = new PyList();
        final PyList temp = py_list.toSet();
        for (int i = 0; i < temp.length() - 1; i++) {
            for (int j = i + 1; j < temp.length(); j++) {
                result.appendsIfAbsent(temp.visits(j) - temp.visits(i));
            }
        }
        return result;
    }

    /**
     * Gives an answer with x0 = 0 and x6 < upper. If no such answer exists, it
     * gives a list consisting of seven -1's.
     * 
     * @param upper an upper bound for x6.
     * @return an answer with x0 = 0 and x6 < upper.
     */
    public static PyList solver1(final int upper) {
        final PyList answer = new PyList();
        for (int x1 = 1; x1 <= upper; x1++) {
            for (int x2 = 1; x2 <= upper; x2++) {
                for (int x3 = 1; x3 <= upper; x3++) {
                    for (int x4 = 1; x4 <= upper; x4++) {
                        for (int x5 = 1; x5 <= upper; x5++) {
                            for (int x6 = 1; x6 <= upper; x6++) {
                                // Sets x0 to be 0, without loss of generality.
                                answer.replacedBy(new PyList(0, x1, x2, x3, x4, x5, x6).toSet());
                                if (difference(answer).length() == 21 && answer.visits(6) < upper) {
                                    return answer;
                                }
                            }
                        }
                    }
                }
            }
        }
        // If no such answer exists, execute the following line.
        return new PyList(-1, -1, -1, -1, -1, -1, -1);
    }

    /**
     * Looks for the minimum solution, given an initial upper bound.
     * 
     * @param arg an initial upper bound.
     */
    public static void solver(final int arg) {
        int upper = arg;
        final PyList answer = new PyList();
        while (true) {
            answer.replacedBy(solver1(upper));
            // Decrease the upper bound.
            upper = answer.visits(6);
            if (upper == -1) {
                break;
            } else {
                // Prints a solution. The last line would be the minimum solution.
                System.out.println(answer);
            }
        }
    }

}

File: PyList.java

package minimum_problem;

/**
 * PyList
 */
public class PyList {

    public static final int MAX_LENGTH = 150;
    private final int[] list;
    private int size;

    /** Initializes the list. */
    public PyList() {
        this.list = new int[PyList.MAX_LENGTH];
        this.resets();
    }

    /**
     * Initializes the list with {@code data}. The elements of {@code data} will be
     * appended to the list.
     * 
     * @param data the elements to be appended.
     */
    public PyList(final int... data) {
        this();
        this.appends(data);
    }

    /**
     * Initializes the list with {@code py_lists}. The elements of {@code py_lists}
     * will be appended to the list.
     * 
     * @param py_lists the lists to be appended.
     */
    public PyList(final PyList... py_lists) {
        this();
        this.appends(py_lists);
    }

    /** Does nothing. */
    public void appends() {

    }

    /**
     * Appends the elements of {@code data} to the list.
     * 
     * @param data the elements to be appended.
     */
    public void appends(final int... data) {
        for (final int datum : data) {
            this.sets(this.length(), datum);
        }
    }

    /**
     * Appends the elements of {@code py_lists} to the list.
     * 
     * @param py_lists the lists to be appended.
     */
    public void appends(final PyList... py_lists) {
        for (final PyList py_list : py_lists) {
            this.appends(py_list.toArray());
        }
    }

    /**
     * Appends {@code element} to the list if it was not in the list.
     * 
     * @param element the element to be appended.
     * @return {@code true} if and only if {@code element} was not in the list.
     */
    public boolean appendsIfAbsent(final int element) {
        if (!this.contains(element)) {
            this.appends(element);
            return true;
        }
        return false;
    }

    /**
     * Validates whether {@code element} is in the list.
     * 
     * @param element the element to be searched.
     * @return {@code true} if and only if {@code element} is in the list.
     */
    public boolean contains(final int element) {
        return this.indexOf(element) != -1;
    }

    @Override
    public boolean equals(final Object other) {
        if (this == other) {
            return true;
        }
        if (other instanceof PyList) {
            final PyList py_list = (PyList) other;
            int n = this.length();
            if (n == py_list.length()) {
                int i = 0;
                while (n-- != 0) {
                    if (this.visits(i) != py_list.visits(i)) {
                        return false;
                    }
                    i++;
                }
                return true;
            }
        }
        return false;
    }

    /**
     * A shortcut to {@code indexOf(element, 0)}.
     * 
     * @param element the element to be searched.
     * @return {@code indexOf(element, 0)}.
     */
    public int indexOf(final int element) {
        return this.indexOf(element, 0);
    }

    /**
     * Looks for the index {@code i} with the properties that {@code beginIndex < i}
     * and that the element at index {@code i} is {@code element}. Gives -1 if no
     * such index exists.
     * 
     * @param element    the element to be searched.
     * @param beginIndex the beginning index.
     * @return the index with the properties above, or -1 if no such index exists.
     */
    public int indexOf(final int element, final int beginIndex) {
        for (int index = beginIndex; index < this.length(); index++) {
            if (this.visits(index) == element) {
                return index;
            }
        }
        return -1;
    }

    /**
     * Gives the length of the list.
     * 
     * @return the length of the size.
     */
    public int length() {
        return this.size;
    }

    /**
     * Gives the last element and removes it.
     * 
     * @return the last element.
     */
    public int pops() {
        return this.pops(this.length() - 1);
    }

    /**
     * Gives the element at index {@code index} and removes it.
     * 
     * @param index the index.
     * @return the element at index {@code index}.
     */
    public int pops(final int index) {
        final PyList temp_list = new PyList();
        final int integer_popped = this.visits(index);
        for (int i = 0; i < this.length(); i++) {
            if (i != index) {
                temp_list.appends(this.visits(i));
            }
        }
        this.replacedBy(temp_list);
        return integer_popped;
    }

    /**
     * Replaces the current list by {@code py_list}.
     * 
     * @param py_list the list.
     */
    public void replacedBy(final PyList py_list) {
        this.resets();
        this.appends(py_list);
    }

    /** Resets the list. */
    public void resets() {
        for (int i = 0; i < PyList.MAX_LENGTH; i++) {
            this.list[i] = 0;
        }
        this.size = 0;
    }

    /**
     * Sets the element at index {@code index} to be {@code element}.
     * 
     * @param index   the index.
     * @param element the element.
     */
    public void sets(final int index, final int element) {
        this.list[index] = element;
        if (index >= this.size) {
            this.size = index + 1;
        }
    }

    /**
     * Sorts the list.
     */
    public void sort() {
        this.replacedBy(this.sorted());
    }

    /**
     * Gives the sorted list of the current one.
     * 
     * @return the sorted list.
     */
    public PyList sorted() {
        final PyList sorted_list = new PyList(this);
        int temporary = 0;
        for (int i = 0; i < sorted_list.length() - 1; i++) {
            for (int j = i + 1; j < sorted_list.length(); j++) {
                if (sorted_list.visits(i) > sorted_list.visits(j)) {
                    temporary = sorted_list.visits(i);
                    sorted_list.sets(i, sorted_list.visits(j));
                    sorted_list.sets(j, temporary);
                }
            }
        }
        return sorted_list;
    }

    /**
     * Converts the list to an array.
     * 
     * @return an array ({@code int[]}).
     */
    public int[] toArray() {
        final int[] array = new int[this.length()];
        for (int i = 0; i < this.length(); i++) {
            array[i] = this.visits(i);
        }
        return array;
    }

    /**
     * Converts the list to a set.
     * 
     * @return a list with no element occurring more than one time.
     */
    public PyList toSet() {
        final PyList set = new PyList();
        for (int i = 0; i < this.length(); i++) {
            set.appendsIfAbsent(this.visits(i));
        }
        return set.sorted();
    }

    @Override
    public String toString() {
        String text = "";
        int index = 0;
        for (; index < this.length() - 1; index++) {
            text += "" + this.visits(index) + " ";
        }
        text += "" + this.visits(index);
        return text;
    }

    /**
     * Visits the element at index {@code index}.
     * 
     * @param index the index.
     * @return the element at index {@code index}.
     */
    public int visits(final int index) {
        return this.list[index];
    }

}

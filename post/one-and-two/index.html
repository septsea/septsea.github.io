package python_list;

/**
 * PyList: Emulates a Python list.
 * </p>
 * To create a new String list called {@code ps}:
 * </p>
 * {@code PyList<String> ps = new PyList<String>();}
 * </p>
 * To create a new list called {@code po}, whose element type is undetermined:
 * </p>
 * {@code PyList<Object> po = new PyList<Object>();}
 */
// The following line suppresses warnings concerning "unchecked".
@SuppressWarnings("unchecked")
public class PyList<E> {

    public static final int MAX_LENGTH = 150;
    private final E[] list;
    private int size;

    /** Initializes the list. */
    public PyList() {
        this.list = (E[]) new Object[MAX_LENGTH];
        this.resets();
    }

    /**
     * Initializes the list with {@code data}. The elements of {@code data} will be
     * appended to the list.
     * 
     * @param data the elements to be appended.
     */
    public PyList(final E... data) {
        this();
        this.appends(data);
    }

    /**
     * Initializes the list with {@code py_lists}. The elements of {@code py_lists}
     * will be appended to the list.
     * 
     * @param py_lists the lists to be appended.
     */
    public PyList(final PyList<E>... py_lists) {
        this();
        this.appends(py_lists);
    }

    /** Does nothing. */
    public void appends() {

    }

    /**
     * Appends the elements of {@code data} to the list.
     * 
     * @param data the elements to be appended.
     */
    public void appends(final E... data) {
        for (final E datum : data) {
            this.sets(this.length(), datum);
        }
    }

    /**
     * Appends the elements of {@code py_lists} to the list.
     * 
     * @param py_lists the lists to be appended.
     */
    public void appends(final PyList<E>... py_lists) {
        for (final PyList<E> py_list : py_lists) {
            for (int i = 0; i < py_list.length(); i++) {
                this.appends(py_list.visits(i));
            }
        }
    }

    /**
     * Appends {@code element} to the list if it was not in the list.
     * 
     * @param element the element to be appended.
     * @return {@code true} if and only if {@code element} was not in the list.
     */
    public boolean appendsIfAbsent(final E element) {
        if (!this.contains(element)) {
            this.appends(element);
            return true;
        }
        return false;
    }

    /**
     * Validates whether {@code element} is in the list.
     * 
     * @param element the element to be searched.
     * @return {@code true} if and only if {@code element} is in the list.
     */
    public boolean contains(final E element) {
        return this.indexOf(element) != -1;
    }

    /**
     * Validates whether the list contains only elements of {@code allowedElements}.
     * 
     * @param allowedElements the array of allowed elements.
     * @return {@code true} if and only if the list contains only elements of
     *         {@code allowedElements}.
     */
    public boolean containsOnly(final E... allowedElements) {
        final PyList<E> allowedElements_list = new PyList<E>(allowedElements);
        for (int i = 0; i < this.length(); i++) {
            if (!allowedElements_list.contains(this.visits(i))) {
                return false;
            }
        }
        return true;
    }

    @Override
    public boolean equals(final Object other) {
        if (this == other) {
            return true;
        }
        if (other instanceof PyList) {
            final PyList<E> py_list = (PyList<E>) other;
            int n = this.length();
            if (n == py_list.length()) {
                int i = 0;
                while (n != 0) {
                    n--;
                    if (this.visits(i) != py_list.visits(i)) {
                        return false;
                    }
                    i++;
                }
                return true;
            }
        }
        return false;
    }

    /**
     * A shortcut to {@code indexOf(element, 0)}.
     * 
     * @param element the element to be searched.
     * @return {@code indexOf(element, 0)}.
     */
    public int indexOf(final E element) {
        return this.indexOf(element, 0);
    }

    /**
     * Looks for the index {@code i} with the properties that {@code beginIndex < i}
     * and that the element at index {@code i} is {@code element}. Gives -1 if no
     * such index exists.
     * 
     * @param element    the element to be searched.
     * @param beginIndex the beginning index.
     * @return the index with the properties above, or -1 if no such index exists.
     */
    public int indexOf(final E element, final int beginIndex) {
        if (element == null) {
            for (int index = beginIndex; index < this.length(); index++) {
                if (this.visits(index) == null) {
                    return index;
                }
            }
        } else {
            for (int index = beginIndex; index < this.length(); index++) {
                if (this.visits(index).equals(element)) {
                    return index;
                }
            }
        }
        return -1;
    }

    /**
     * Inserts {@code element} before {@code index}.
     * 
     * @param index   the index.
     * @param element the element to be inserted.
     */
    public void inserts(final int index, final E element) {
        if (index >= this.length()) {
            this.appends(element);
        } else {
            int i = this.length();
            while (i != index) {
                i--;
                this.sets(i + 1, this.visits(i));
            }
            this.sets(index, element);
        }
    }

    /**
     * Gives the length of the list.
     * 
     * @return the length of the size.
     */
    public int length() {
        return this.size;
    }

    /**
     * Gives the last element and removes it.
     * 
     * @return the last element.
     */
    public E pops() {
        return this.pops(this.length() - 1);
    }

    /**
     * Gives the element at index {@code index} and removes it.
     * 
     * @param index the index.
     * @return the element at index {@code index}.
     */
    public E pops(final int index) {
        final PyList<E> temp_list = new PyList<E>();
        final E integer_popped = this.visits(index);
        for (int i = 0; i < this.length(); i++) {
            if (i != index) {
                temp_list.appends(this.visits(i));
            }
        }
        this.replacedBy(temp_list);
        return (E) integer_popped;
    }

    /**
     * Replaces the current list by {@code py_list}.
     * 
     * @param py_list the list.
     */
    public void replacedBy(final PyList<E> py_list) {
        this.resets();
        this.appends(py_list);
    }

    /** Resets the list. */
    public void resets() {
        for (int i = 0; i < PyList.MAX_LENGTH; i++) {
            this.list[i] = null;
        }
        this.size = 0;
    }

    /**
     * Sets the element at index {@code index} to be {@code element}.
     * 
     * @param index   the index.
     * @param element the element.
     */
    public void sets(final int index, final E element) {
        this.list[index] = element;
        if (index >= this.size) {
            this.size = index + 1;
        }
    }

    /**
     * Gives the sorted list of the given one.
     * 
     * @param py_list the list to be sorted.
     * @return the sorted list.
     */
    public static PyList<Integer> sorted(final PyList<Integer> py_list) {
        final PyList<Integer> sorted_list = new PyList<Integer>(py_list);
        int temporary = 0;
        for (int i = 0; i < sorted_list.length() - 1; i++) {
            for (int j = i + 1; j < sorted_list.length(); j++) {
                if (sorted_list.visits(i) > sorted_list.visits(j)) {
                    temporary = sorted_list.visits(i);
                    sorted_list.sets(i, sorted_list.visits(j));
                    sorted_list.sets(j, temporary);
                }
            }
        }
        return sorted_list;
    }

    /**
     * Appends all elements of the list to a given array. Because of a bug, an array
     * E[] must be passed as a parameter. Here is an example:
     * </p>
     * {@code PyList<String> k = new PyList<String>("1a", "2b");}
     * </p>
     * {@code System.out.println(k.toArray(new String[k.length()])[1]);}
     * </p>
     * This prints {@code 2b}.
     * 
     * @param array the array to be appended.
     * @return an array consisting of elements of the list.
     */
    public E[] toArray(final E[] array) {
        for (int i = 0; i < this.length(); i++) {
            array[i] = this.visits(i);
        }
        return array;
    }

    /**
     * Converts the list to a set.
     * 
     * @return a list with no element occurring more than one time.
     */
    public PyList<E> toSet() {
        final PyList<E> set = new PyList<E>();
        for (int i = 0; i < this.length(); i++) {
            set.appendsIfAbsent(this.visits(i));
        }
        return set;
    }

    @Override
    public String toString() {
        String text = "[";
        int index = 0;
        for (; index < this.length() - 1; index++) {
            text += "" + this.visits(index) + ", ";
        }
        text += "" + this.visits(index) + "]";
        return text;
    }

    /**
     * Visits the element at index {@code index}.
     * 
     * @param index the index.
     * @return the element at index {@code index}.
     */
    public E visits(final int index) {
        return this.list[index];
    }

}

package one_and_two;

/**
 * GCD
 */
public class GCD {

    /**
     * Gives a Bézout expression.
     * 
     * @param a the first integer.
     * @param b the second integer.
     * @return the expression "au + bv = gcd(a, b)" in the string form.
     */
    public static String Bézout(int a, int b) {
        int[] t = GCD.Bézout_implementation(a, b);
        return String.format("%d * %d + %d * %d = %d", a, t[0], b, t[1], t[2]);
    }

    /**
     * Finds a pair of integers u, v such that au + bv = gcd(a, b).
     * 
     * @param a the first integer.
     * @param b the second integer.
     * @return { u, v, gcd(a, b) }.
     */
    public static int[] Bézout_implementation(final int a, final int b) {
        // Set f = { a, b, -1 }. It is not hard to see
        // inner_product(f0, t0) = 0
        // inner_product(f0, t1) = 0
        // Since all the 3D vectors are linear combinations of t0 and t1
        // the result is also orthogonal to f
        // The euclidean algorithm shows that the third component
        // of the result is the GCD (greatest common divisor)
        // of a and b
        final int[] t0 = { 1, 0, a };
        final int[] t1 = { 0, 1, b };
        final int[] temp = { 1, 0, a };
        int q = 0;
        int r = 0;
        while (t1[2] != 0) {
            temp[0] = t0[0];
            temp[1] = t0[1];
            temp[2] = t0[2];
            t0[0] = t1[0];
            t0[1] = t1[1];
            t0[2] = t1[2];
            q = GCD.quotient(temp[2], t0[2]);
            r = GCD.remainder(temp[2], t0[2]);
            t1[0] = temp[0] - t0[0] * q;
            t1[1] = temp[1] - t0[1] * q;
            t1[2] = r;
        }
        // The GCD is usually nonnegative.
        if (t0[2] < 0) {
            t0[0] *= -1;
            t0[1] *= -1;
            t0[2] *= -1;
        }
        return t0;
    }

    /**
     * Gives the quotient.
     * 
     * @param a the dividend.
     * @param b the divisor.
     * @return the quotient.
     * @throws ArithmeticException if b is zero.
     */
    public static int quotient(int a, int b) throws ArithmeticException {
        if (b == 0) {
            throw new ArithmeticException("Division by zero");
        }
        return GCD.division(a, b)[0];
    }

    /**
     * Gives the remainder.
     * 
     * @param a the dividend.
     * @param b the divisor.
     * @return the remainder.
     * @throws ArithmeticException if b is zero.
     */
    public static int remainder(int a, int b) throws ArithmeticException {
        if (b == 0) {
            throw new ArithmeticException("Division by zero");
        }
        return GCD.division(a, b)[1];
    }

    /**
     * Performs division.
     * 
     * @param a the dividend.
     * @param b the divisor.
     * @return a 2-array consisting of the quotient and remainder.
     * @throws ArithmeticException if b is zero.
     */
    public static int[] division(final int a, final int b) throws ArithmeticException {
        if (b == 0) {
            throw new ArithmeticException("Division by zero");
        } else {
            // Computes |a| / |b| and |a| % |b| first.
            int quotient = Math.abs(a) / Math.abs(b);
            int remainder = Math.abs(a) % Math.abs(b);
            // Suppose a, b > 0, with a = bq + r, 0 <= r < b. Then
            // a = (-b)(-q) + r
            // If r = 0, -a = b(-q) = (-b)q
            // If r > 0,
            // -a = b(-q) - r = b(-q) + (b - r) - b = b(-q-1) + (b - r)
            // -a = (-b)(q+1) + (-(-b) - r)
            //
            // If a >= 0, this part is ignored
            if (a < 0) {
                // If remainder = 0, this part is ignored
                if (remainder > 0) {
                    remainder = Math.abs(b) - remainder;
                    quotient = -quotient - 1;
                }
            }
            // If b > 0, this part is ignored
            if (b < 0) {
                quotient *= -1;
            }
            final int[] result = { quotient, remainder };
            return result;
        }
    }

}

public class Main {

    public static void main(String... args) {
        System.out.println("\nProblem:");
        System.out.println("Find the least positive integer with the following properties:");
        System.out.println("    (a) 23 divides l;");
        System.out.println("    (b) 29 divides l + 1;");
        System.out.println("    (c) The decimal expression of l contains only digits 1 and 2;");
        System.out.println("    (d) The decimal expression of l + 1 contains only digits 1 and 2.");
        System.out.println("\nSolution using some math:");
        System.out.println("Let the integer be l. Then there exist two integers p, q such that");
        System.out.println("    29q = 23p + 1    (l = 23p)");
        System.out.println("A moment's thought gives");
        System.out.println("    " + GCD.Bézout(-23, 29));
        System.out.println("Hence");
        System.out.println("    p = 5 + 29t              q = 4 + 23t");
        System.out.println("Since l consists of dights 1 and 2, l % 10 must be 1. Then");
        System.out.println("    23(5 + 29t) = 10r + 1");
        System.out.println("    10r - 667t = 114");
        System.out.println("A moment's thought gives");
        System.out.println("    " + GCD.Bézout(10, -667));
        System.out.println("Hence");
        System.out.println("    r = -200 * 114 + 667s    t = -3 * 114 + 10s");
        System.out.println("    r = 667s - 22800         t = 10s - 342");
        System.out.println("    l = 23p = 23(5 + 29t) = 23(290s - 9913) > 0");
        System.out.println("Then s > 9913/290. Then");
        System.out.println("    s >= 35");
        System.out.println("Put s = 35        -->        l = 5451");
        System.out.println("Put s = 36        -->        l = 12121");
        System.out.println("Hence l = 12121. That is all.\n");
        long time = 0;
        time = System.currentTimeMillis();
        System.out.println("Brute force solution:");
        Main.solve_by_brute_force();
        time = System.currentTimeMillis() - time;
        System.out.println("\nTime elapsed: " + time + (time < 2 ? " millisecond." : " milliseconds.") + "\n");
    }

    public static boolean containsNoMoreThan(String text, String... allowedCharacters) {
        PyList<String> a = new PyList<String>();
        for (int i = 0; i < text.length(); i++) {
            a.appendsIfAbsent("" + text.charAt(i));
        }
        return a.containsOnly(allowedCharacters);
    }

    public static void solve_by_brute_force() {
        int l = 0;
        while (true) {
            if (l % 23 != 0) {
                l++;
                continue;
            }
            if ((l + 1) % 29 != 0) {
                l++;
                continue;
            }
            if (!Main.containsNoMoreThan("" + l, "1", "2")) {
                l++;
                continue;
            }
            if (!Main.containsNoMoreThan("" + (l + 1), "1", "2")) {
                l++;
                continue;
            } else {
                System.out.println(l + " / 23 = " + l / 23);
                System.out.println((l + 1) + " / 29 = " + (l + 1) / 29);
                break;
            }
        }
    }
